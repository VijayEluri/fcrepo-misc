243     @Test
244     public void dsvBinaryContent() throws IOException {
245         final String content = "A word starting with c, followed by o, "
246                 + "followed by n, followed by t, followed by e, "
247                 + "followed by n, and ending with t.";
248         File tempFile = File.createTempFile("fcrepo-dto-test", null);
249         FileOutputStream sink = null;
250         try {
251             //
252             // Write test
253             //
254             sink = new FileOutputStream(tempFile);
255             IOUtils.copy(new StringReader(content),
256                     new FileOutputStream(tempFile));
257             IOUtils.closeQuietly(sink);
258             Datastream ds = new Datastream("ds").controlGroup(
259                     ControlGroup.MANAGED);
260             ds.addVersion(null).contentLocation(tempFile.toURI());
261             DatastreamVersion origDSV = ds.versions().first();
262             obj.putDatastream(ds);
263             // serialize obj to FOXML, requesting that the managed content is
264             // embedded as binaryContent (base64), and ensure the resulting
265             // XML is exactly as expected.
266             writeCheck("dsvBinaryContent", obj.datastreams().keySet());
267             //
268             // Read test
269             //
270             FOXMLReader reader = new FOXMLReader();
271             FedoraObject result = readCheck("dsvBinaryContent", reader, false);
272             DatastreamVersion resultDSV = result.datastreams().get("ds").versions().first();
273             // should have new file: contentLocation
274             URI resultLocation = resultDSV.contentLocation();
275             Assert.assertNotNull(resultLocation);
276             Assert.assertFalse(resultLocation.equals(origDSV.contentLocation()));
277             Assert.assertEquals(resultLocation.getScheme(), "file");
278             // ...whose content matches that of the original
279             File resultFile = new File(resultLocation.getRawSchemeSpecificPart());
280             Assert.assertTrue(resultFile.exists());
281             FileInputStream resultIn = new FileInputStream(resultFile);
282             String resultContent = IOUtils.toString(resultIn, "UTF-8");
283             IOUtils.closeQuietly(resultIn);
284             Assert.assertEquals(resultContent, content);
285             // other than the contentLocation, the newly-read object should
286             // be equivalent to the original one
287             resultDSV.contentLocation(origDSV.contentLocation());
288             Assert.assertEquals(obj, result);
289             // when the reader is closed, it should delete resultFile
290             reader.close();
291             Assert.assertFalse(resultFile.exists());
292         } finally {
293             IOUtils.closeQuietly(sink);
294             tempFile.delete();
295         }
296     }

355     private void writeCheck(String testName, Set<String> embedIds) {
356         String filename = "foxml/" + testName + ".xml";
357         try {
358             byte[] actualBytes = getFOXML(embedIds).getBytes("UTF-8");
359             String actualXML = new String(actualBytes, "UTF-8");
360             if (getClass().getClassLoader().getResource(filename) == null) {
361                 if (testResources != null && testResources.isDirectory()) {
362                     File testFile = new File(testResources, filename);
363                     testFile.getParentFile().mkdirs();
364                     OutputStream out = new FileOutputStream(testFile);
365                     IOUtils.copy(new StringReader(actualXML), out);
366                     IOUtils.closeQuietly(out);
367                     System.out.println("Generated test resource: " + filename);
368                 } else {
369                     Assert.fail("Test resource not found: " + filename);
370                 }
371             } else {
372                 InputStream expectedStream = getClass().getClassLoader()
373                         .getResourceAsStream(filename);
374                 byte[] expectedBytes = XMLUtil.prettyPrint(
375                         IOUtils.toByteArray(expectedStream), false);
376                 String expectedXML = new String(expectedBytes, "UTF-8");
377                 Assert.assertEquals(expectedXML, actualXML);
378             }
379         } catch (IOException e) {
380             throw new RuntimeException(e);
381         }
382     }

384     private FedoraObject readCheck(String testName, FOXMLReader reader,
385                                    boolean check) {
386         String filename = "foxml/" + testName + ".xml";
387         try {
388             InputStream in = getClass().getClassLoader()
389                     .getResourceAsStream(filename);
390             FedoraObject result = reader.readObject(in);
391             if (check) {
392                 Assert.assertEquals(obj, result);
393             }
394             return result;
395         } catch (IOException e) {
396             throw new RuntimeException(e);
397         }
398     }

400     private String getFOXML(Set<String> embedIds) {
401         try {
402             FOXMLWriter writer = new FOXMLWriter();
403             writer.setManagedDatastreamsToEmbed(embedIds);
404             ByteArrayOutputStream out = new ByteArrayOutputStream();
405             writer.writeObject(obj, out);
406             byte[] pretty = XMLUtil.prettyPrint(out.toByteArray(), false);
407             return new String(pretty, "UTF-8");
408         } catch (IOException e) {
409             throw new RuntimeException(e);
410         }
