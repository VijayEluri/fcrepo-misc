54      @Override
55      public FedoraObject readObject(InputStream source) throws IOException {
56          obj = new FedoraObject();
57          XMLInputFactory factory = XMLInputFactory.newInstance();
58          try {
59              r = factory.createXMLStreamReader(source, Constants.CHAR_ENCODING);
60              readObject();
61              return obj;
62          } catch (XMLStreamException e) {
63              throw new IOException(e);
64          } finally {
65              XMLUtil.closeQuietly(r);
66              IOUtils.closeQuietly(source);
67          }
68      }

70      private void readObject() throws IOException, XMLStreamException {
71          if (moveToStart(Constants.digitalObject, null)) {
72              obj.pid(readAttribute(Constants.PID));
73              readObjectProperties();
74              while (r.getEventType() == XMLStreamConstants.START_ELEMENT
75                      && r.getLocalName().equals(Constants.datastream)) {
76                  readDatastream();
77              }
78          }
79      }

81      private void readObjectProperties() throws XMLStreamException {
82          while (moveToStart(Constants.property, Constants.datastream)) {
83              String name = readAttribute(Constants.NAME);
84              String value = readAttribute(Constants.VALUE);
85              if (name != null) {
86                  if (name.equals(Constants.STATE_URI)) {
87                      obj.state(parseState(value, "object"));
88                  } else if (name.equals(Constants.LABEL_URI)) {
89                      obj.label(value);
90                  } else if (name.equals(Constants.OWNERID_URI)) {
91                      obj.ownerId(value);
92                  } else if (name.equals(Constants.CREATEDDATE_URI)) {
93                      obj.createdDate(parseDate(value, "object created"));
94  
95                  } else if (name.equals(Constants.LASTMODIFIEDDATE_URI)) {
96                      obj.lastModifiedDate(parseDate(value,
97                              "object last modified"));
98                  } else {
99                      logger.warn("Ignoring unrecognized object property name: "
100                             + name);
101                 }
102             }
103         }
104     }

106     private void readDatastream() throws IOException, XMLStreamException {
107         String id = readAttribute(Constants.ID);
108         if (id != null) {
109             Datastream ds = new Datastream(id);
110             obj.putDatastream(ds);
111             ds.state(parseState(readAttribute(Constants.STATE), "datastream"));
112             ds.controlGroup(parseControlGroup(readAttribute(
113                     Constants.CONTROL_GROUP)));
114             ds.versionable(parseVersionable(readAttribute(
115                     Constants.VERSIONABLE)));
116             while (moveToStart(Constants.datastreamVersion,
117                     Constants.datastream)) {
118                 readDatastreamVersion(ds);
119             }
120         } else {
121             logger.warn("Ignoring datastream; no id specified");
122         }
123     }

125     private void readDatastreamVersion(Datastream ds)
126             throws IOException, XMLStreamException {
127         String id = readAttribute(Constants.ID);
128         if (id != null) {
129             Date created = parseDate(readAttribute(Constants.CREATED),
130                     "datastream created");
131             DatastreamVersion dsv = new DatastreamVersion(id, created);
132             ds.versions().add(dsv);
133             dsv.altIds().addAll(parseAltIds(readAttribute(Constants.ALT_IDS)));
134             dsv.label(readAttribute(Constants.LABEL));
135             dsv.mimeType(readAttribute(Constants.MIMETYPE));
136             dsv.formatURI(parseURI(readAttribute(Constants.FORMAT_URI),
137                     "datastream format uri"));
138             dsv.size(parseLong(readAttribute(Constants.SIZE),
139                     "datastream size"));
140             if (r.nextTag() == XMLStreamConstants.START_ELEMENT) {
141                 if (r.getLocalName().equals(Constants.contentDigest)) {
142                     readContentDigest(dsv);
143                     if (r.nextTag() == XMLStreamConstants.END_ELEMENT) {
144                         return; // end of datastreamVersion
145                     }
146                 }
147                 if (r.getLocalName().equals(Constants.xmlContent)) {
148                     readXMLContent(dsv);
149                 } else if (r.getLocalName().equals(Constants.binaryContent)) {
150                     readBinaryContent(ds, dsv);
151                 } else if (r.getLocalName().equals(Constants.contentLocation)) {
152                     readContentLocation(dsv);
153                 }
154             }
155         } else {
156             logger.warn("Ignoring datastream version; no id specified");
157         }
158     }

160     private void readContentLocation(DatastreamVersion dsv)
161             throws XMLStreamException {
162         String type = readAttribute(Constants.TYPE);
163         URI ref = parseURI(readAttribute(Constants.REF), "contentLocation ref");
164         if (ref != null) {
165             if (Constants.INTERNALREF_TYPE.equals(type)) {
166                 try {
167                     dsv.contentLocation(new URI(Constants.INTERNALREF_SCHEME
168                             + ":" + ref));
169                 } catch (URISyntaxException e) {
170                     logger.warn("Ignoring malformed contentLocation "
171                             + "internal ref: " + ref);
172                 }
173             } else {
174                 dsv.contentLocation(ref);
175             }
176         }
177     }

179     private void readBinaryContent(Datastream ds, DatastreamVersion dsv)
180             throws IOException, XMLStreamException {
181         OutputStream sink = contentHandler.handleContent(obj, ds, dsv);
182         if (sink == null) return; // handler opted out
183         OutputStream out = null;
184         try {
185             out = new Base64OutputStream(sink, false);
186             while (r.next() != XMLStreamConstants.END_ELEMENT) {
187                 if (r.isCharacters()) {
188                     out.write(r.getText().getBytes(Constants.CHAR_ENCODING));
189                 }
190             }
191             out.flush();
192         } finally {
193             IOUtils.closeQuietly(out);
194             IOUtils.closeQuietly(sink);
195         }
196     }

198     private void readXMLContent(DatastreamVersion dsv)
199             throws IOException, XMLStreamException {
200         while (r.next() != XMLStreamConstants.START_ELEMENT) {
201             if (r.getEventType() == XMLStreamConstants.END_ELEMENT) {
202                 return; // xmlContent element is empty
203             }
204         }
205         ByteArrayOutputStream sink = new ByteArrayOutputStream();
206         try {
207             XMLUtil.copy(r, sink);
208             dsv.inlineXML(new InlineXML(sink.toByteArray()));
209         } catch (Exception e) {
210             throw new IOException("Error parsing foxml:xmlContent", e);
211         }
212     }

214     private void readContentDigest(DatastreamVersion dsv)
215             throws XMLStreamException {
216         String type = readAttribute(Constants.TYPE);
217         String digest = readAttribute(Constants.DIGEST);
218         if (type != null || digest != null) {
219             dsv.contentDigest(new ContentDigest().type(type).hexValue(digest));
220         }
221         r.nextTag(); // consume closing contentDigest tag
222     }

224     private boolean moveToStart(String localName,
225                                 String stopAtLocalName)
226             throws XMLStreamException {
227         while (r.hasNext()) {
228             if (r.next() == XMLStreamConstants.START_ELEMENT) {
229                 if (r.getLocalName().equals(localName)) {
230                     return true;
231                 } else if (r.getLocalName().equals(stopAtLocalName)) {
232                     return false;
233                 }
234             }
235         }
236         return false;
237     }

239     private String readAttribute(String localName) throws XMLStreamException {
240         String value = r.getAttributeValue(null, localName);
241         if (value != null && value.trim().length() > 0) {
242             return value.trim();
243         } else {
244             return null;
245         }
246     }

248     private static Date parseDate(String value, String kind) {
249         if (value == null) return null;

283     private static Boolean parseVersionable(String value) {
284         if (value == null) return null;
285         if (value.equalsIgnoreCase("true") || value.equals("1")) {
286             return true;
287             
288         } else if (value.equalsIgnoreCase("false") || value.equals("0")) {
289             return false;
290         } else {
291             logger.warn("Ignoring unrecognized datastream versionable value: "
292                     + value);
293             return null;
294         }

317     private static Set<URI> parseAltIds(String value) {
318         Set<URI> set = new HashSet<URI>();
319         if (value != null) {
320             for (String uriString: value.split("\\s+")) {
321                 URI uri = parseURI(uriString, "datastream altId");
322                 if (uri != null) set.add(uri);
323             }
324         }
325         return set;
