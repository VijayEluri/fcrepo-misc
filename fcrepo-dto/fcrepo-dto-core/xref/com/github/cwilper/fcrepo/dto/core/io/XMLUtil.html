72      /**
73       * Pretty-prints the given XML.
74       * <p>
75       * The input must be well-formed. The transformation has the following
76       * effect:
77       * <ul>
78       *   <li> Leading and trailing whitespace will be removed.</li>
79       *   <li> All processing instructions and doctype declarations will be
80       *        removed.</li>
81       *   <li> If Xalan is the currently active XML transformer (true by
82       *        default in Java 6), it will be reformatted to have two-space
83       *        indents. Otherwise, indenting behavior is undefined (it may or
84       *        may not be indented).</li>
85       *   <li> Attributes will use double-quotes around values.</li>
86       *   <li> All empty elements (e.g.
87       *        <code>&lt;element&gt;&lt;element&gt;</code>) will be
88       *        collapsed (e.g. <code>&lt;element/&gt;</code></li>
89       * </ul>
90       *
91       * @param inBytes the xml to be pretty-printed.
92       * @param omitXMLDeclaration whether to omit the xml declaration in the
93       *        output. If false, the output will not be directly embeddable in
94       *        other XML documents.
95       * @return the pretty-printed XML as a UTF-8 encoded byte array.
96       * @throws IOException
97       */
98      public static byte[] prettyPrint(byte[] inBytes,
99                                       boolean omitXMLDeclaration)
100             throws IOException {
101         ByteArrayOutputStream sink = new ByteArrayOutputStream();
102         prettyPrint(new ByteArrayInputStream(inBytes),
103                 new OutputStreamWriter(sink, "UTF-8"), omitXMLDeclaration);
104         return sink.toByteArray();
105     }

107     private static void prettyPrint(InputStream source,
108                                     Writer sink,
109                                     boolean omitXMLDeclaration)
110             throws IOException {
111         try {
112             TransformerFactory tFactory = TransformerFactory.newInstance();
113             String xsl = prettyXSL;
114             if (omitXMLDeclaration) xsl = prettyXSLLite;
115             Transformer t = tFactory.newTransformer(
116                     new StreamSource(new StringReader(xsl)));
117             t.setErrorListener(new DebugLoggingErrorListener());
118             StringWriter preTrimmed = new StringWriter();
119             t.transform(new StreamSource(source), new StreamResult(preTrimmed));
120             preTrimmed.flush();
121             sink.write(preTrimmed.toString().trim());
122             sink.flush();
123         } catch (Exception e) {
124             throw new IOException(e);
125         } finally {
126             IOUtils.closeQuietly(source);
127         }
128     }

130     /**
131      * Produces the canonicalized form of the input document, without comments,
132      * as defined by <a href="http://www.w3.org/TR/xml-c14n11/">XML
133      * Canonicalization 1.1</a>.
134      * <p>
135      * <b>NOTE:</b> Although this method will likely fail if the input is
136      * malformed XML, successful canonicalization does not imply that the
137      * input was well-formed!
138      *
139      * @param inBytes the xml to be canonicalized.
140      * @return the canonicalized XML as a UTF-8 encoded byte array.
141      * @throws IOException if the input cannot be canonicalized for any reason.
142      */
143     public static byte[] canonicalize(byte[] inBytes) throws IOException {
144         try {
145             Canonicalizer c = Canonicalizer.getInstance(
146                     Canonicalizer.ALGO_ID_C14N11_OMIT_COMMENTS);
147             return c.canonicalize(inBytes);
148         } catch (Exception e) {
149             throw new IOException(e);
150         }
151     }

185     /**
186      * Copies the subtree that the reader is currently position at to the
187      * given stream. The reader must be positioned at a START_ELEMENT
188      * event, and when finished, will be positioned on the corresponding
189      * END_ELEMENT event.
190      * <p>
191      * <b>NOTE:</b> Even if successful, this does not currently ensure that
192      * the output has all used namespaces declared. In particular, if a
193      * namespace was previously declared in the XML that the reader saw,
194      * then used within the scope of the elements being copied but no
195      * re-declared within them, the output will not be self-contained.
196      *
197      * @param source the reader whose subtree should be copied.
198      * @param sink the stream to write to (UTF-8 encoding will be used).
199      *        It will be left open when finished.
200      * @throws XMLStreamException if anything goes wrong during the copy.
201      */
202     public static void copy(XMLStreamReader source, OutputStream sink)
203             throws XMLStreamException {
204         XMLOutputFactory factory = XMLOutputFactory.newInstance();
205         XMLStreamWriter writer = factory.createXMLStreamWriter(sink, "UTF-8");
206         copy(source, writer);
207         writer.flush();
208     }
