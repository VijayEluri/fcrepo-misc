9   /**
10   * The inline XML content of a {@link DatastreamVersion}.
11   * <p>
12   * <h2>Canonicalization and Normalization</h2>
13   * The XML given at construction time must be well-formed.  It will be
14   * automatically <i>canonicalized</i> according to
15   * <a href="http://www.w3.org/TR/xml-c14n11/">Canonical XML Version 1.1</a>
16   *(c14n11). In the rare event that the XML is well-formed but canonicalization
17   * is impossible, (for example, if a namespace declaration in the XML has a
18   * relative URI) it will be <i>normalized</i> using the following non-standard
19   * algorithm:
20   * <ol>
21   *   <li> Leading and trailing whitespace will be removed.</li>
22   *   <li> All XML declaration, processing instructions, and doctype
23   *        declarations will be removed. This ensures it can be
24   *        embedded within an XML document.</li>
25   *   <li> If Xalan is the currently active XML transformer (true by default in
26   *        Java 6), it will be reformatted to have two-space indents. Otherwise,
27   *        indenting behavior is undefined (it may or may not be indented).</li>
28   *   <li> Attributes will use double-quotes around values.</li>
29   *   <li> All empty elements (e.g.
30   *        <code>&lt;element&gt;&lt;element&gt;</code>) will be
31   *        collapsed (e.g. <code>&lt;element/&gt;</code></li>
32   * </ol>
33   * If standard canonicalization (c14n11) is successful at construction time,
34   * {@link #canonical()} will always return <code>true</code>.
35   */

47       * Creates an instance from a string.

50       * @throws IOException if the value is not a well-formed, standalone
51       *         XML document.
52       */
53      public InlineXML(String value) throws IOException {
54          this(Util.getBytes(value));
55      }

57      /**
58       * Creates an instance from a UTF-8 encoded byte array.
59       *
60       * @param bytes a well-formed, standalone XML document.
61       * @throws IOException if the value is not a well-formed, standalone
62       *         XML document or cannot be decoded as UTF-8.
63       */
64      public InlineXML(byte[] bytes) throws IOException {
65          // verify it's well-formed first; canonicalization doesn't
66          byte[] normalized = XMLUtil.prettyPrint(bytes, true);
67          // try to canonicalize, otherwise, use normalized form
68          try {
69              this.bytes = XMLUtil.canonicalize(bytes);
70              this.canonical = true;
71          } catch (IOException e) {
72              logger.warn("Unable to canonicalize (c14n11); using non-standard "
73                      + "normalization instead", e);
74              this.bytes = normalized;
75          }
76          this.value = Util.getString(this.bytes);
77      }
78  
79      /**
80       * Gets the canonicalized (or normalized) XML as a string.
81       *
82       * @return the value, never <code>null</code>.
83       */
84      public String value() {
85          return value;
86      }
87  
88      /**
89       * Gets the canonicalized (or normalized) XML as a UTF-8 byte array.
90       *
91       * @return the value, never <code>null</code>.
92       */
93      public byte[] bytes() {
94          return bytes;
95      }

100      * @return true if it was successfully converted to c14n11 form at
101      *         construction time, false otherwise. If false, the caller can
102      *         assume the XML is in the non-standard normalized form
103      *         described above.
104      */
