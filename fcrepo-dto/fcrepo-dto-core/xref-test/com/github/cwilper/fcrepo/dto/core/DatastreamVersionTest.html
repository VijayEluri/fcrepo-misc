69      public void createdDateField() {
70          DatastreamVersion dsv;
71          Date date;
72          // set value, get same value
73          date = new Date(0);
74          dsv = new DatastreamVersion("a", date);
75          Assert.assertEquals(date, dsv.createdDate());
76          // changing the original date object after setting it
77          // doesn't affect the value stored
78          date.setTime(1);
79          Assert.assertEquals(0, dsv.createdDate().getTime());
80          // changing the retrieved date object after getting it
81          // shouldn't affect the value stored
82          date = dsv.createdDate();
83          date.setTime(1);
84          Assert.assertEquals(0, dsv.createdDate().getTime());
85          // set null, get null
86          dsv = new DatastreamVersion("a", null);
87          Assert.assertNull(dsv.createdDate());
