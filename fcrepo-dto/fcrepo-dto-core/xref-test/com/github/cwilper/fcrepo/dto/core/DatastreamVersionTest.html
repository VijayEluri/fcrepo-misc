67      public void createdDateField() {
68          DatastreamVersion dsv;
69          Date date;
70          // set value, get same value
71          date = new Date(0);
72          dsv = new DatastreamVersion("a", date);
73          Assert.assertEquals(date, dsv.createdDate());
74          // changing the original date object after setting it
75          // doesn't affect the value stored
76          date.setTime(1);
77          Assert.assertEquals(0, dsv.createdDate().getTime());
78          // changing the retrieved date object after getting it
79          // shouldn't affect the value stored
80          date = dsv.createdDate();
81          date.setTime(1);
82          Assert.assertEquals(0, dsv.createdDate().getTime());
83          // set null, get null
84          dsv = new DatastreamVersion("a", null);
85          Assert.assertNull(dsv.createdDate());

151     @Test
152     public void canonicalizableInlineXML() throws IOException {
153         DatastreamVersion dsv = new DatastreamVersion("a", null);
154         // value starts null
155         Assert.assertNull(dsv.inlineXML());
156         Assert.assertNull(dsv.inlineXMLBytes());
157         // canonicalized starts false
158         Assert.assertFalse(dsv.inlineXMLCanonicalized());
159         // set value, get canonicalized value
160         dsv.inlineXML("<doc/>");
161         Assert.assertEquals("<doc></doc>", dsv.inlineXML());
162         Assert.assertEquals("<doc></doc>",
163                 Util.getString(dsv.inlineXMLBytes()));
164         // canonicalized should now be true
165         Assert.assertTrue(dsv.inlineXMLCanonicalized());
166         // set null, get null
167         dsv.inlineXML(null);
168         Assert.assertNull(dsv.inlineXML());
169         Assert.assertNull(dsv.inlineXMLBytes());
170         // canonicalized should now be false
171         Assert.assertFalse(dsv.inlineXMLCanonicalized());
172     }
