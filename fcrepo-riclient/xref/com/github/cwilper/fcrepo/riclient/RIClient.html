45      private InputStream executeQuery(String query,
46                                       String lang,
47                                       boolean flush) {
48          HttpPost post = new HttpPost(riEndpoint);
49          List<NameValuePair> args = new ArrayList<NameValuePair>();
50          if (flush) {
51              args.add(new BasicNameValuePair("flush", "true"));
52          }
53          args.add(new BasicNameValuePair("type", "tuples"));
54          args.add(new BasicNameValuePair("lang", lang));
55          args.add(new BasicNameValuePair("format", "Simple"));
56          args.add(new BasicNameValuePair("stream", "on"));
57          args.add(new BasicNameValuePair("query", query));
58          try {
59              post.setEntity(new UrlEncodedFormEntity(args, HTTP.UTF_8));
60          } catch (UnsupportedEncodingException wontHappen) {
61              throw new RuntimeException(wontHappen);
62          }
63          try {
64              HttpResponse response = httpClient.execute(post);
65              int responseCode = response.getStatusLine().getStatusCode();
66              if (responseCode != 200) {
67                  throw new RuntimeException("Fedora Resource Index query "
68                          + "returned an unexpected HTTP response code: "
69                          + responseCode + ". Consult Fedora Server log for "
70                          + "details.");
71              }
72              HttpEntity entity = response.getEntity();
73              if (entity == null) {
74                  return new ByteArrayInputStream(new byte[0]);
75              } else {
76                  return entity.getContent();
77              }
78          } catch (IOException e) {
79              throw new RuntimeException(e);
80          }
81      }
