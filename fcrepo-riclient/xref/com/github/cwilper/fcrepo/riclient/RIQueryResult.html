22   * The result of a Resource Index query.
23   */
24  public class RIQueryResult
25          extends AbstractSource<List<Value>> {
26  
27      private static final Logger logger =
28              LoggerFactory.getLogger(RIQueryResult.class);
29  
30      private final BufferedReader reader;
31  
32      private final ValueFactory factory;
33  
34      private boolean exhausted;
35  
36      RIQueryResult(InputStream in) {
37          try {
38              reader = new BufferedReader(new InputStreamReader(in, HTTP.UTF_8));
39          } catch (UnsupportedEncodingException wontHappen) {
40              throw new RuntimeException(wontHappen);
41          }
42          factory = new ValueFactoryImpl();
43          exhausted = false;
44      }
45  
46      @Override
47      protected List<Value> computeNext() throws IOException {
48          if (exhausted) {
49              return endOfData();
50          } else {
51              List<Value> values = new ArrayList<Value>();
52              String line = reader.readLine();
53              while (line != null && line.length() > 0) {
54                  values.add(parse(line));
55                  line = reader.readLine();
56              }
57              if (values.size() == 0) {
58                  return endOfData();
59              } else if (line == null) {
60                  exhausted = true;
61                  close();
62              }
63              return values;
64          }
65      }
66  
67      @PreDestroy
68      @Override
69      public void close() {
70          try {
71              reader.close();
72          } catch (IOException e) {
73              logger.warn("Error closing reader", e);
74          }
75      }
76  
77      // Parse a query result value of the form "name : NTriplesValue"
78      private Value parse(String line) {
79          int i = line.indexOf(" : ");
80          if (i == -1) {
81              throw new IllegalArgumentException("Malformed line in RI Query "
82                      + "result (expected ' : ' delimiter): '" + line + "'");
83          } else {
84              return NTriplesUtil.parseValue(line.substring(i + 3), factory);
85          }
86      }
